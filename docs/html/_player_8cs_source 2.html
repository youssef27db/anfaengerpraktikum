121 private void HandleMovement(double DeltaTime) {
122 Vector2 direction = new Vector2(Input.GetAxis("ui_left", "ui_right"), Input.GetAxis("ui_up", "ui_down")).Normalized();
123 float currentSpeed = SPEED;
124
125 // Sprite umdrehen basierend auf der Bewegungsrichtung und Kollision umdrehen
126 if (direction.X < 0) {
127 Sprite.FlipH = true;
128 SwordCollision.Position = new Vector2(-Mathf.Abs(SwordCollision.Position.X), SwordCollision.Position.Y);
129 PlayerHitbox.Position = new Vector2(Sprite.Position.X * 1.8f, PlayerHitbox.Position.Y);
130 } else if (direction.X > 0) {
131 Sprite.FlipH = false;
132 SwordCollision.Position = new Vector2(Mathf.Abs(SwordCollision.Position.X), SwordCollision.Position.Y);
133 PlayerHitbox.Position = HauptHitbox;
134 }
135
136 // Geschwindigkeit reduzieren, wenn der Spieler angreift
137 if (AnimationPlayer.CurrentAnimation == "light_attack") {
138 currentSpeed *= 0.5f;
139 } else if (AnimationPlayer.CurrentAnimation == "heavy_attack") {
140 currentSpeed *= 0.15f;
141 }
142
143 // Blockieren stoppt die Bewegung
144 if (IsBlocking()) {
145 currentSpeed = 0;
146 }
147
148 if (IsDashing) {
149 DashInProgress(DeltaTime);
150 } else {
151 // Normale Bewegung verarbeiten, wenn kein Dash aktiv ist
152 if (direction != Vector2.Zero) {
153 Velocity = new Vector2(direction.X * currentSpeed, Velocity.Y);
154 } else {
155 Velocity = new Vector2(Mathf.MoveToward(Velocity.X, 0, SPEED), Velocity.Y);
156 }
157
158 // Überprüfen, ob der Dash-Button gedrückt wurde mit eine Bewegungsrichtung und nicht schon am angreifen ist
159 if (Input.IsActionJustPressed("dash") && direction != Vector2.Zero && CanDash && !IsAttacking()) {
160 // Wenn der Player genug Stamina hat kann er dashen
161 if (UseStamina(20)){
162 DashDirection = direction;
163 StartDash();
164 }
165 }
166 }
