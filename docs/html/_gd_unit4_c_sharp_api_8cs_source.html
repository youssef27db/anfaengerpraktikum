106 // Convert a set of Tests stored as Dictionaries to TestSuiteNode
107 // all tests are assigned to a single test suit
108 internal static TestSuiteNode BuildTestSuiteNodeFrom(Array<Dictionary> tests)
109 {
110 if (tests.Count == 0)
111 throw new InvalidOperationException("Cant build 'TestSuiteNode' from an empty test set.");
112
113 // Create a suite ID
114 var suiteId = Guid.NewGuid();
115 var firstTest = tests[0];
116 var managedType = firstTest["managed_type"].AsString();
117 var assemblyLocation = firstTest["assembly_location"].AsString();
118 var sourceFile = firstTest["source_file"].AsString();
119
120 // Create TestCaseNodes for each test in the suite
121 var testCaseNodes = tests
122 .Select(test => new TestCaseNode
123 {
124 Id = Guid.Parse(test["guid"].AsString()),
125 ParentId = suiteId,
126 ManagedMethod = test["test_name"].AsString(),
127 LineNumber = test["line_number"].AsInt32(),
128 AttributeIndex = test["attribute_index"].AsInt32(),
129 RequireRunningGodotEngine = test["require_godot_runtime"].AsBool()
130 }
131 )
132 .ToList();
133
134 return new TestSuiteNode
135 {
136 Id = suiteId,
137 ParentId = Guid.Empty,
138 ManagedType = managedType,
139 AssemblyPath = assemblyLocation,
140 SourceFile = sourceFile,
141 Tests = testCaseNodes
142 };
143 }
