150 private void HandleMovement(double DeltaTime){
151 if(Dead) return;
152 if((Sprite.Animation == "take_hit" || Sprite.Animation == "attack") && Sprite.IsPlaying()){
153 Velocity = Vector2.Zero;
154 return;
155 }
156 if(Pursuing){
157 AnimationState = State.WALK;
158 TargetPosition = CurrentTarget.Position;
159 if(IsCloseTo(Position.X, TargetPosition.X, 0.1f)){
160 AnimationState = State.IDLE;
161 Velocity = Vector2.Zero;
162 return;
163 }
164 ReturnToStart = ReturnToStartAfter;
165 } else if(ReturnToStart >= 0){
166 AnimationState = State.IDLE;
167 ReturnToStart -= DeltaTime;
168 TargetPosition = Vector2.Inf;
169 } else if(!IsCloseTo(Position.X, StartPosition.X, 0.1f)){
170 AnimationState = State.WALK;
171 TargetPosition = StartPosition;
172 }
173
174 if(TargetPosition != Vector2.Inf){
175
176 if(IsCloseTo(Position.X, TargetPosition.X, 0.1f)){
177 AnimationState = State.IDLE;
178 Velocity = Vector2.Zero;
179 if(TargetPosition == StartPosition && Sprite.FlipH != StartRotation){
180 FlipRotation();
181 }
182 TargetPosition = Vector2.Inf;
183 return;
184 }
185
186 if(TargetPosition.X > Position.X){
187 SetRotation(true);
188 if(!FrontCollisionRayCast.IsColliding()){
189 Vector2 velocity = Vector2.Zero;
190 velocity.X = Speed;
191 Velocity = velocity;
192 }
193 } else {
194 SetRotation(false);
195 if(!FrontCollisionRayCast.IsColliding()){
196 Vector2 velocity = Vector2.Zero;
197 velocity.X = -Speed;
198 Velocity = velocity;
199 }
200 }
201
202 if((!RightFallProtection.IsColliding() && !Sprite.FlipH) || (!LeftFallProtection.IsColliding() && Sprite.FlipH)){
203 Velocity = Vector2.Zero;
204 }
205
206 } else {
207 Velocity = Vector2.Zero;
208 AnimationState = State.IDLE;
209 }
